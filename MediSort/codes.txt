  <script>
    $(document).ready(function() {
      // Symptom card selection
      $('.symptom-card').click(function() {
        $(this).toggleClass('active');
        updateHiddenInputs();
      });

      // Form submission
      $('#triageForm').submit(function(e) {
        e.preventDefault();
        
        const formData = {
          Age: $('#age').val(),
          Duration: $('#duration').val(),
          'Primary Symptom': $('#primarySymptom').val(),
          'Additional Symptoms': $('#additionalSymptoms').val(),
          Severity: $('#severity').val(),
          'Pain Location': $('#painLocation').val(),
          Allergies: $('#allergies').val()
        };

        // Simple triage logic based on symptoms
        setTimeout(() => {
          const primarySymptom = $('#primarySymptom').val().toLowerCase();
          const severity = parseInt($('#severity').val());
          
          let response;
          
          // Emergency symptoms
          if (['chest pain', 'shortness of breath', 'severe bleeding'].includes(primarySymptom) || severity >= 9) {
            response = {
              triage_category: "Emergency Care Needed",
              confidence: 0.90,
              recommended_action: "Seek emergency care immediately",
              probabilities: {
                "Emergency": 0.85,
                "Urgent": 0.10,
                "Non-urgent": 0.05
              }
            };
          }
          // Urgent symptoms
          else if (['fever', 'abdominal pain', 'headache'].includes(primarySymptom) || severity >= 7) {
            response = {
              triage_category: "Urgent Care Recommended",
              confidence: 0.80,
              recommended_action: "Visit urgent care within 24 hours",
              probabilities: {
                "Emergency": 0.20,
                "Urgent": 0.60,
                "Non-urgent": 0.20
              }
            };
          }
          // Non-urgent symptoms
          else {
            response = {
              triage_category: "Non-urgent Care",
              confidence: 0.75,
              recommended_action: "Schedule a doctor's appointment",
              probabilities: {
                "Emergency": 0.05,
                "Urgent": 0.20,
                "Non-urgent": 0.75
              }
            };
          }
          
          displayResults(response);
        }, 1000);
      });

      function updateHiddenInputs() {
        // Duration
        const duration = $('.symptom-card.active[data-value]').first().data('value');
        $('#duration').val(duration);
        
        // Additional symptoms
        const symptoms = [];
        $('.symptom-card.active').each(function() {
          symptoms.push($(this).data('value'));
        });
        $('#additionalSymptoms').val(symptoms.join(', '));
      }

      function displayResults(data) {
        $('#triageResult').text(data.triage_category);
        $('#recommendedAction').text(data.recommended_action);
        $('#confidenceLevel').text((data.confidence * 100).toFixed(0) + '% confident');
        $('#confidenceBar').css('width', (data.confidence * 100) + '%');
        
        let probabilitiesHtml = '<div class="mt-3"><h5>Detailed Probabilities:</h5><ul class="list-group">';
        for (const [category, prob] of Object.entries(data.probabilities)) {
          probabilitiesHtml += `
            <li class="list-group-item d-flex justify-content-between align-items-center">
              ${category}
              <span class="badge badge-primary badge-pill">${(prob * 100).toFixed(0)}%</span>
            </li>`;
        }
        probabilitiesHtml += '</ul></div>';
        $('#probabilityDetails').html(probabilitiesHtml);
        
        $('#resultContainer').fadeIn();
        $('html, body').animate({
          scrollTop: $('#resultContainer').offset().top - 100
        }, 500);
      }
    });
  </script>

    $(document).ready(function() {
      // Symptom card selection
      $('.symptom-card').click(function() {
        $(this).toggleClass('active');
        updateHiddenInputs();
      });

      // Form submission
      $('#triageForm').submit(function(e) {
        e.preventDefault();
        
        const formData = {
          Age: $('#age').val(),
          Duration: $('#duration').val(),
          'Primary Symptom': $('#primarySymptom').val(),
          'Additional Symptoms': $('#additionalSymptoms').val(),
          Severity: $('#severity').val(),
          'Pain Location': $('#painLocation').val(),
          Allergies: $('#allergies').val()
        };

        // Simple triage logic based on symptoms
        setTimeout(() => {
          const primarySymptom = $('#primarySymptom').val().toLowerCase();
          const severity = parseInt($('#severity').val());
          
          let response;
          
          // Emergency symptoms
          if (['chest pain', 'shortness of breath', 'severe bleeding'].includes(primarySymptom) || severity >= 9) {
            response = {
              triage_category: "Emergency Care Needed",
              confidence: 0.90,
              recommended_action: "Seek emergency care immediately",
              probabilities: {
                "Emergency": 0.85,
                "Urgent": 0.10,
                "Non-urgent": 0.05
              }
            };
          }
          // Urgent symptoms
          else if (['fever', 'abdominal pain', 'headache'].includes(primarySymptom) || severity >= 7) {
            response = {
              triage_category: "Urgent Care Recommended",
              confidence: 0.80,
              recommended_action: "Visit urgent care within 24 hours",
              probabilities: {
                "Emergency": 0.20,
                "Urgent": 0.60,
                "Non-urgent": 0.20
              }
            };
          }
          // Non-urgent symptoms
          else {
            response = {
              triage_category: "Non-urgent Care",
              confidence: 0.75,
              recommended_action: "Schedule a doctor's appointment",
              probabilities: {
                "Emergency": 0.05,
                "Urgent": 0.20,
                "Non-urgent": 0.75
              }
            };
          }
          
          displayResults(response);
        }, 1000);
      });

      function updateHiddenInputs() {
        // Duration
        const duration = $('.symptom-card.active[data-value]').first().data('value');
        $('#duration').val(duration);
        
        // Additional symptoms
        const symptoms = [];
        $('.symptom-card.active').each(function() {
          symptoms.push($(this).data('value'));
        });
        $('#additionalSymptoms').val(symptoms.join(', '));
      }

      function displayResults(data) {
        $('#triageResult').text(data.triage_category);
        $('#recommendedAction').text(data.recommended_action);
        $('#confidenceLevel').text((data.confidence * 100).toFixed(0) + '% confident');
        $('#confidenceBar').css('width', (data.confidence * 100) + '%');
        
        let probabilitiesHtml = '<div class="mt-3"><h5>Detailed Probabilities:</h5><ul class="list-group">';
        for (const [category, prob] of Object.entries(data.probabilities)) {
          probabilitiesHtml += `
            <li class="list-group-item d-flex justify-content-between align-items-center">
              ${category}
              <span class="badge badge-primary badge-pill">${(prob * 100).toFixed(0)}%</span>
            </li>`;
        }
        probabilitiesHtml += '</ul></div>';
        $('#probabilityDetails').html(probabilitiesHtml);
        
        $('#resultContainer').fadeIn();
        $('html, body').animate({
          scrollTop: $('#resultContainer').offset().top - 100
        }, 500);
      }
    });

              probabilities: {
                "Emergency": 0.05,

                "Non-urgent": 0.75
              }
            };

  <script>
    $(document).ready(function() {
      // Symptom card selection
      $('.symptom-card').click(function() {
        $(this).toggleClass('active');
        updateHiddenInputs();
      });

      // Form submission
      $('#triageForm').submit(function(e) {
        e.preventDefault();
        
        const formData = {
          Age: $('#age').val(),
          Duration: $('#duration').val(),
          'Primary Symptom': $('#primarySymptom').val(),
          'Additional Symptoms': $('#additionalSymptoms').val(),
          Severity: $('#severity').val(),
          'Pain Location': $('#painLocation').val(),
          Allergies: $('#allergies').val()
        };

          // Enhanced triage logic based on symptoms
          setTimeout(() => {
            const primarySymptom = $('#primarySymptom').val().toLowerCase();
            const severity = parseInt($('#severity').val());
            const additionalSymptoms = $('#additionalSymptoms').val().toLowerCase();
            
            let response;
            
            // Emergency symptoms (life-threatening conditions)
            if (['chest pain', 'shortness of breath', 'severe bleeding', 'seizures', 'loss of consciousness', 'severe head injury'].includes(primarySymptom) || 
                severity >= 9 ||
                (additionalSymptoms.includes('difficulty breathing') && severity >= 7)) {
              response = {
                triage_category: "Emergency Care Needed",
                recommended_action: "Seek emergency care immediately",
                reasoning: "Your symptoms suggest a potentially life-threatening condition that requires immediate medical evaluation."
              };
            }
            // Urgent symptoms (should be seen within 24 hours)
          
          displayResults(response);
        }, 1000);
      });

      function updateHiddenInputs() {
        // Duration
        const duration = $('.symptom-card.active[data-value]').first().data('value');
        $('#duration').val(duration);
        
        // Additional symptoms
        const symptoms = [];
        $('.symptom-card.active').each(function() {
          symptoms.push($(this).data('value'));
        });
        $('#additionalSymptoms').val(symptoms.join(', '));
      }

      function displayResults(data) {
        $('#triageResult').text(data.triage_category);
        $('#recommendedAction').text(data.recommended_action);
        
        // Clear previous probability display
        $('#confidenceLevel').remove();
        $('#confidenceBar').parent().remove();
        
        // Show reasoning in a clean format
        $('#probabilityDetails').html(`
          <div class="reasoning-box">
            <h5>Recommendation Details:</h5>
            <p>${data.reasoning}</p>
          </div>
        `);
        
        // Style based on urgency
        const resultBox = $('#resultContainer');
        resultBox.removeClass('emergency urgent home');
        if (data.triage_category.includes('Emergency')) {
          resultBox.addClass('emergency');
        } else if (data.triage_category.includes('Urgent')) {
          resultBox.addClass('urgent');
        } else {
          resultBox.addClass('home');
        }
        
        $('#resultContainer').fadeIn();
        $('html, body').animate({
          scrollTop: $('#resultContainer').offset().top - 100
        }, 500);
      }
    });
  </script>

    $(document).ready(function() {
      // Symptom card selection
      $('.symptom-card').click(function() {
        $(this).toggleClass('active');
        updateHiddenInputs();
      });

      // Form submission
      $('#triageForm').submit(function(e) {
        e.preventDefault();
        
        const formData = {
          Age: $('#age').val(),
          Duration: $('#duration').val(),
          'Primary Symptom': $('#primarySymptom').val(),
          'Additional Symptoms': $('#additionalSymptoms').val(),
          Severity: $('#severity').val(),
          'Pain Location': $('#painLocation').val(),
          Allergies: $('#allergies').val()
        };

          // Enhanced triage logic based on symptoms
          setTimeout(() => {
            const primarySymptom = $('#primarySymptom').val().toLowerCase();
            const severity = parseInt($('#severity').val());
            const additionalSymptoms = $('#additionalSymptoms').val().toLowerCase();
            
            let response;
            
            // Emergency symptoms (life-threatening conditions)
            if (['chest pain', 'shortness of breath', 'severe bleeding', 'seizures', 'loss of consciousness', 'severe head injury', 'sudden paralysis', 'severe allergic reaction'].includes(primarySymptom) || 
                severity >= 9 ||
                (additionalSymptoms.includes('difficulty breathing') && severity >= 7) ||
                (additionalSymptoms.includes('chest pain') && severity >= 7)) {
              response = {
                triage_category: "Emergency Care Needed",
                recommended_action: "Seek emergency care immediately",
                reasoning: "Your symptoms suggest a potentially life-threatening condition that requires immediate medical evaluation."
              };
            }
            // Urgent symptoms (should be seen within 24 hours)
            else if (['fever', 'abdominal pain', 'headache', 'dizziness', 'vomiting', 'diarrhea', 'rash with fever', 'eye pain', 'severe pain'].includes(primarySymptom) ||
                severity >= 7 ||
                (additionalSymptoms.includes('fever') && severity >= 5) ||
                (additionalSymptoms.includes('vomiting') && severity >= 5) ||
                (additionalSymptoms.includes('diarrhea') && ['Less than 1 day', '1-2 days'].includes($('#duration').val()))) {
              response = {
                triage_category: "Urgent Care Recommended",
                recommended_action: "Visit urgent care within 24 hours",
                reasoning: "Your symptoms require prompt medical attention but are not immediately life-threatening."
              };
            }
            // Home care (can wait for regular appointment)
            else {
              response = {
                triage_category: "Home Care Recommended",
                recommended_action: "Schedule a doctor's appointment if symptoms persist",
                reasoning: "Your symptoms appear to be mild and can likely be managed with self-care. Monitor your condition and seek medical attention if symptoms worsen or persist."
              };
            }
          
          displayResults(response);
        }, 1000);
      });

      function updateHiddenInputs() {
        // Duration
        const duration = $('.symptom-card.active[data-value]').first().data('value');
        $('#duration').val(duration);
        
        // Additional symptoms
        const symptoms = [];
        $('.symptom-card.active').each(function() {
          symptoms.push($(this).data('value'));
        });
        $('#additionalSymptoms').val(symptoms.join(', '));
      }

      function displayResults(data) {
        $('#triageResult').text(data.triage_category);
        $('#recommendedAction').text(data.recommended_action);
        
        // Clear previous probability display
        $('#confidenceLevel').remove();
        $('#confidenceBar').parent().remove();
        
        // Show reasoning in a clean format
        $('#probabilityDetails').html(`
          <div class="reasoning-box">
            <h5>Recommendation Details:</h5>
            <p>${data.reasoning}</p>
          </div>
        `);
        
        // Style based on urgency
        const resultBox = $('#resultContainer');
        resultBox.removeClass('emergency urgent home');
        if (data.triage_category.includes('Emergency')) {
          resultBox.addClass('emergency');
        } else if (data.triage_category.includes('Urgent')) {
          resultBox.addClass('urgent');
        } else {
          resultBox.addClass('home');
        }
        
        $('#resultContainer').fadeIn();
        $('html, body').animate({
          scrollTop: $('#resultContainer').offset().top - 100
        }, 500);
      }
    });
  </script>

      // Initialize variables
      let currentStep = 1;
      const totalSteps = 4;
      let selectedSymptoms = [];
      
      // Update triage level as user fills form
      function updateTriageLevel() {
        let level = 0;
        
        // Age factor
        const age = parseInt($('#age').val()) || 0;
        if (age > 60 || age < 5) level += 10;
        
        // Primary symptom factor
        const primarySymptom = $('#primarySymptom').val().toLowerCase();
        if (['chest pain', 'shortness of breath', 'severe bleeding', 'seizures'].includes(primarySymptom)) {
          level += 30;
        } else if (primarySymptom) {
          level += 10;
        }
        
        // Additional symptoms factor
        level += selectedSymptoms.length * 5;
        
        // Severity factor
        const severity = parseInt($('#severity').val()) || 5;
        level += (severity - 1) * 3;
        
        // Cap at 100
        level = Math.min(100, Math.max(0, level));
        
        // Update indicator
        $('#triageProgress').css('width', level + '%');
        $('#triageLevelText').text(
          level < 30 ? 'Low' : 
          level < 60 ? 'Moderate' : 'High'
        );
        
        // Show/hide indicator
        if (level > 0) {
          $('#triageIndicator').fadeIn();
        } else {
          $('#triageIndicator').fadeOut();
        }
      }

        goToStep(prevStep);
      });

          selectedSymptoms.push(symptomValue);

          selectedSymptoms.splice(index, 1);
        }

    function updateTriageLevel() {
      let level = 0;
        
        // Age factor (more weight for very young and elderly)
      const age = parseInt($('#age').val()) || 0;
      if (age > 75 || age < 2) level += 20;
      else if (age > 65 || age < 5) level += 15;
      else if (age > 55 || age < 12) level += 10;
        
        // Primary symptom factor
      const primarySymptom = $('#primarySymptom').val().toLowerCase();
      const emergencySymptoms = ['chest pain', 'shortness of breath', 'severe bleeding', 'loss of consciousness'];
      const urgentSymptoms = ['severe abdominal pain', 'fever above 103', 'palpitations', 'burning while urinating'];
      
      if (emergencySymptoms.some(s => primarySymptom.includes(s))) {
        level += 40;
      } else if (urgentSymptoms.some(s => primarySymptom.includes(s))) {
        level += 25;
      } else if (primarySymptom) {
        level += 10;
      }
        
        // Additional symptoms factor - weighted by importance
      const highRiskSymptoms = ['fever', 'chest pain', 'dizziness', 'shortness of breath'];
      let additionalSymptomScore = 0;
      
      selectedSymptoms.forEach(symptom => {
        if (highRiskSymptoms.some(s => symptom.toLowerCase().includes(s))) {
          additionalSymptomScore += 8;
        } else {
          additionalSymptomScore += 3;
        }
      });
      
      level += Math.min(additionalSymptomScore, 30); // Cap at 30 points
        
         // Severity factor
      const severity = parseInt($('#severity').val()) || 5;
      level += (severity - 1) * 3;
        // Duration factor
      const duration = $('#duration').val() || '';
      if (duration === 'Less than 1 day' && severity > 7) {
        level += 10; // Acute severe symptoms
      } else if (duration.includes('weeks')) {
        level += 5; // Chronic symptoms
      }
        // Cap at 100
        level = Math.min(100, Math.max(0, level));

      $('#triageProgress').css('width', level + '%');

        $('#triageLevelText').text('Low');
      } else if (level < 60) {
        $('#triageLevelText').text('Moderate');
      } else if (level < 85) {
        $('#triageLevelText').text('Urgent');

    function updateTriageLevel() {
  let level = 0;
  const age = parseInt($('#age').val()) || 0;
  const primarySymptom = $('#primarySymptom').val().toLowerCase();
  const duration = $('#duration').val();

  //  Age factor
  if (age > 75 || age < 2) level += 25;
  else if (age > 65 || age < 5) level += 30;
  else if (age > 55 || age < 12) level += 15;

  //  Primary symptom
  const emergencySymptoms = ['chest pain', 'shortness of breath', 'severe bleeding', 'loss of consciousness'];
  const urgentSymptoms = ['severe abdominal pain','frequent urination', 'fever above 103', 'palpitations', 'burning while urinating'];

  if (emergencySymptoms.some(sym => primarySymptom.includes(sym))) {
    level += 60;
  } else if (urgentSymptoms.some(sym => primarySymptom.includes(sym))) {
    level += 45;
  } else if (primarySymptom.length > 0) {
    level += 20;
  }

  //  Additional symptoms (simulate selection list  placeholder logic)
  const additionalSymptoms = []; //  use your real symptom selection list if you have it
  const highRiskSymptoms = ['fever', 'chest pain', 'dizziness', 'shortness of breath'];
  let additionalScore = 0;
  additionalSymptoms.forEach(sym => {
    if (highRiskSymptoms.some(h => sym.toLowerCase().includes(h))) {
      additionalScore += 10;
    } else {
      additionalScore += 4;
    }
  });
  level += Math.min(additionalScore, 40);

  //  Duration
  if (duration === 'Less than 1 day') level += 10;
  else if (duration.includes('week')) level += 30;

  // Cap and apply level
  level = Math.min(100, Math.max(0, level));
  $('#triageProgress').css('width', level + '%');

  let levelText = 'Low';
  if (level >= 60) levelText = 'High';
  else if (level >= 30) levelText = 'Moderate';
  $('#triageLevelText').text(levelText);

  if (level > 0) $('#triageIndicator').fadeIn();
  else $('#triageIndicator').fadeOut();

  return level;
}
